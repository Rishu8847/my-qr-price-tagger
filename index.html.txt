    <script>
        let html5QrCode;
        let currentScanResult = null;

        // Initialize and Start Scanner
        function startScanner() {
            html5QrCode = new Html5Qrcode("reader");
            const config = { fps: 10, qrbox: { width: 250, height: 250 } };

            html5QrCode.start({ facingMode: "environment" }, config, (decodedText) => {
                onScanSuccess(decodedText);
            });
            document.getElementById('scan-btn').classList.add('hidden');
        }

        function onScanSuccess(decodedText) {
            currentScanResult = decodedText;
            
            // UI Updates
            document.getElementById('result-panel').classList.remove('hidden');
            document.getElementById('qr-data-text').innerText = decodedText;
            
            // Stop scanner to save battery/focus
            html5QrCode.stop();
            document.getElementById('scan-btn').classList.remove('hidden');

            // Data Logic: Check if price exists
            const savedPrices = JSON.parse(localStorage.getItem('qrPrices') || '{}');
            const inputField = document.getElementById('price-input');
            
            if (savedPrices[decodedText]) {
                inputField.value = savedPrices[decodedText];
                alert("Match found! Existing price loaded.");
            } else {
                inputField.value = "";
                inputField.focus();
            }
        }

        // Save Data to LocalStorage
        function savePrice() {
            const price = document.getElementById('price-input').value;
            if (!price) return alert("Please enter a price");

            const savedPrices = JSON.parse(localStorage.getItem('qrPrices') || '{}');
            savedPrices[currentScanResult] = price;
            
            localStorage.setItem('qrPrices', JSON.stringify(savedPrices));
            alert("Price saved successfully!");
            renderHistory();
        }

        // Render History List
        function renderHistory() {
            const list = document.getElementById('history-list');
            const savedPrices = JSON.parse(localStorage.getItem('qrPrices') || '{}');
            list.innerHTML = '';

            Object.entries(savedPrices).reverse().forEach(([data, price]) => {
                const item = document.createElement('div');
                item.className = "bg-white p-4 rounded-xl shadow-sm border border-gray-100 flex justify-between items-center";
                item.innerHTML = `
                    <div class="overflow-hidden mr-4">
                        <p class="text-xs text-gray-400 truncate w-40">${data}</p>
                        <p class="font-bold text-gray-800">${price}</p>
                    </div>
                    <button onclick="deletePrice('${data}')" class="text-red-400 hover:text-red-600 text-sm">Delete</button>
                `;
                list.appendChild(item);
            });
        }

        function deletePrice(key) {
            const savedPrices = JSON.parse(localStorage.getItem('qrPrices') || '{}');
            delete savedPrices[key];
            localStorage.setItem('qrPrices', JSON.stringify(savedPrices));
            renderHistory();
        }

        // Initial Load
        renderHistory();
    </script>
